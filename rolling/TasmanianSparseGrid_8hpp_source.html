1757  DomainInsideSignature getDomainInside() const{
1758  auto rule = getRule();
1759  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1760  return [](std::vector<double> const &)->bool{ return true; };
1761  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1762  if (domain_transform_a.empty()){
1763  return [](std::vector<double> const &x)->bool{
1764  for(auto const &v : x) if (v < 0.0) return false;
1765  return true;
1766  };
1767  }else{
1768  size_t dims = (size_t) getNumDimensions();
1769  return [=](std::vector<double> const &x)->bool{
1770  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1771  return true;
1772  };
1773  }
1774  }else{
1775  if (domain_transform_a.empty()){
1776  if (isFourier()){
1777  return [](std::vector<double> const &x)->bool{
1778  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1779  return true;
1780  };
1781  }else{
1782  return [](std::vector<double> const &x)->bool{
1783  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1784  return true;
1785  };
1786  }
1787  }else{
1788  size_t dims = (size_t) getNumDimensions();
1789  return [=](std::vector<double> const &x)->bool{
1790  for(size_t i=0; i<dims; i++)
1791  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1792  return true;
1793  };
1794  }
1795  }
